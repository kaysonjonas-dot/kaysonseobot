async function fetchUrl(url) { // 優先使用 Cloudflare Worker 代理 try { const proxyUrl = 'https://steep-dust-b2bf.kaysonjonas.workers.dev/?url=' + encodeURIComponent(url); const response = await fetch(proxyUrl); if (response.ok) { return await response.text(); } } catch (e) { console.log('Cloudflare Worker proxy failed, trying fallback...'); } // 嘗試直接抓取（僅適用無 CORS 限制的網站） try { const response = await fetch(url, { mode: 'cors', headers: { 'Accept': 'text/html' } }); if (response.ok) { return await response.text(); } } catch (e) {} // 嘗試使用 allorigins 代理 try { const proxyUrl = 'https://api.allorigins.win/raw?url=' + encodeURIComponent(url); const response = await fetch(proxyUrl); if (response.ok) { return await response.text(); } } catch (e) {} alert('無法抓取該網址。請使用「貼上 HTML」方式進行分析。\n\n錯誤：所有代理服務均無法存取該網站。'); return ''; }
