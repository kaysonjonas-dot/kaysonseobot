function analyzeMeta(doc) { const title = doc.querySelector('title'); const titleText = title ? title.textContent.trim() : ''; // 加強版 description 偵測（大小寫不敏感 + 多種屬性） let description = ''; const allMetas = doc.querySelectorAll('meta'); // 方法1：標準 meta description（大小寫不敏感） for (const meta of allMetas) { const name = meta.getAttribute('name'); if (name && name.toLowerCase() === 'description') { description = meta.getAttribute('content') || ''; break; } } // 方法2：如果沒找到，嘗試 Open Graph description if (!description) { for (const meta of allMetas) { const property = meta.getAttribute('property'); if (property && property.toLowerCase() === 'og:description') { description = meta.getAttribute('content') || ''; break; } } } // 方法3：嘗試 Twitter Card description if (!description) { for (const meta of allMetas) { const name = meta.getAttribute('name'); if (name && name.toLowerCase() === 'twitter:description') { description = meta.getAttribute('content') || ''; break; } } } const titleLength = titleText.length; const descLength = description.length; return { title: { content: titleText, length: titleLength, status: getStatus(titleLength, STANDARDS.title), exists: !!titleText }, description: { content: description, length: descLength, status: getStatus(descLength, STANDARDS.description), exists: !!description } }; }
